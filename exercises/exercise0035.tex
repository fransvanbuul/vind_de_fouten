\begin{opgave}
Je gaat tennisschoenen kopen en bekijkt de prijzen in 5 verschillende sportwinkels:

\begin{itemize}
\item Sport2000: \texteuro{89}, \texteuro{125}, \texteuro{67}
\item Decathlon: \texteuro{45}, \texteuro{78}, \texteuro{92}, \texteuro{156}
\item Intersport: \texteuro{134}, \texteuro{98}
\item Tennis Plaza: \texteuro{178}, \texteuro{203}, \texteuro{145}, \texteuro{167}, \texteuro{189}
\item Sportwinkel De Bal: \texteuro{76}, \texteuro{112}
\end{itemize}

Wat is de gemiddelde schoenprijs per winkel? Welke winkel heeft de laagste
gemiddelde prijs en hoeveel scheelt dat met de duurste gemiddelde prijs?
\end{opgave}

\begin{oplossing}
Gemiddelde prijzen per winkel:
Sport2000: (89 + 125 + 67) \textrm{\textdiv} 3 = 281 \textrm{\textdiv} 3 = \texteuro{93,67}
Decathlon: (45 + 78 + 92 + 156) \textrm{\textdiv} 4 = 371 \textrm{\textdiv} 4 = \texteuro{92,75}
Intersport: (134 + 98) \textrm{\textdiv} 2 = 232 \textrm{\textdiv} 2 = \texteuro{116,00}
Tennis Plaza: (178 + 203 + 145 + 167 + 189) \textrm{\textdiv} 5 = 882 \textrm{\textdiv} 5 = \texteuro{176,40}
De Bal: (76 + 112) \textrm{\textdiv} 2 = 188 \textrm{\textdiv} 2 = \texteuro{94,00}

Laagste: Decathlon (\texteuro{92,75})
Hoogste: Tennis Plaza (\texteuro{176,40})
Verschil: \texteuro{176,40} - \texteuro{92,75} = \texteuro{83,65}

Antwoord: Decathlon laagste (\texteuro{92,75}), verschil \texteuro{83,65}
\end{oplossing}
