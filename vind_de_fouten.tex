  \fi

% Environment for multiple-choice questions
\newenvironment{vragen}{%
  \par\medskip
  \textbf{Vragen:}
  \par\smallskip
}{%
  \par\medskip
}

% Command for a single question
% Usage: \vraag{line number}{text fragment}{option A}{option B}{option C}{option D}
\newcounter{vraagnummer}
\newcommand{\vraag}[6]{%
  \stepcounter{vraagnummer}
  \par\medskip
  \noindent\textbf{Vraag \thevraagnummer} (regel #1): ``#2''
  \par\smallskip
  \begin{enumerate}[label=\Alph*., leftmargin=1.5em]
    \item #3
    \item #4
    \item #5
    \item #6
  \end{enumerate}
}
